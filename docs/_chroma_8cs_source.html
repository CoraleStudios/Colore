242  [SuppressMessage("StyleCop.CSharp.NamingRules", "SA1305:FieldNamesMustNotUseHungarianNotation",
243  Justification = "Parameter names match those in the Message struct.")]
244  public bool HandleMessage(IntPtr handle, int msgId, IntPtr wParam, IntPtr lParam)
245  {
246  if (!_registered)
247  throw new InvalidOperationException("Register must be called before event handling can be performed.");
248 
249  if (handle != _registeredHandle)
250  {
251  throw new ArgumentException(
252  "The specified window handle does not match the currently registered one.",
253  "handle");
254  }
255 
256  if (msgId != Constants.WmChromaEvent)
257  return false;
258 
259  var handled = false;
260 
261  var type = wParam.ToInt32();
262  var state = lParam.ToInt32();
263 
264  switch (type)
265  {
266  case 1: // Chroma SDK support
267  OnSdkSupport(state != 0);
268  handled = true;
269  break;
270 
271  case 2: // Access to device
272  OnDeviceAccess(state != 0);
273  handled = true;
274  break;
275 
276  case 3: // Application state
277  OnApplicationState(state != 0);
278  handled = true;
279  break;
280  }
281 
282  return handled;
283  }
