138  static NativeMethods()
139  {
140  IntPtr chromaSdkPointer;
141 
142  if (EnvironmentHelper.Is64BitProcess() && EnvironmentHelper.Is64BitOperatingSystem())
143  {
144  // We are running 64-bit!
145  chromaSdkPointer = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK64.dll");
146  }
147  else
148  {
149  // We are running 32-bit!
150  chromaSdkPointer = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK.dll");
151  }
152 
153  if (chromaSdkPointer == IntPtr.Zero)
154  {
155  throw new ColoreException(
156  "Failed to dynamically load Chroma SDK library (Error " + Marshal.GetLastWin32Error() + ").");
157  }

458  private static T GetDelegateFromLibrary<T>(IntPtr lib, string name)
459  {
460  var functionPtr = Native.Kernel32.NativeMethods.GetProcAddress(lib, name);
461  if (functionPtr == IntPtr.Zero)
462  {
463  throw new ColoreException(
464  "Failed to dynamically load function, GetProcAddress returned NULL for " + name);
465  }
466 
467  return (T)(object)Marshal.GetDelegateForFunctionPointer(functionPtr, typeof(T));
468  }
