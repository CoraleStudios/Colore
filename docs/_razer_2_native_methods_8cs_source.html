161  static NativeMethods()
162  {
163  if (EnvironmentHelper.Is64BitProcess() && EnvironmentHelper.Is64BitOperatingSystem())
164  {
165  // We are running 64-bit!
166  ChromaSdkPointer = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK64.dll");
167  SystemArchitecture = Architecture.X64;
168  }
169  else
170  {
171  // We are running 32-bit!
172  ChromaSdkPointer = Native.Kernel32.NativeMethods.LoadLibrary("RzChromaSDK.dll");
173  SystemArchitecture = Architecture.X86;
174  }
175 
176  if (ChromaSdkPointer == IntPtr.Zero)

178  throw new ColoreException(

479  private static T GetDelegateFromLibrary<T>(IntPtr lib, string name)
480  {
481  var functionPtr = Native.Kernel32.NativeMethods.GetProcAddress(lib, name);
482  if (functionPtr == IntPtr.Zero)
483  {
484  throw new ColoreException(
485  "Failed to dynamically load function, GetProcAddress returned NULL for " + name);
486  }
487 
488  return (T)(object)Marshal.GetDelegateForFunctionPointer(functionPtr, typeof(T));
489  }
