480  internal static DeviceInfo QueryDevice(Guid id)
481  {
482  var ptr = Marshal.AllocHGlobal(Marshal.SizeOf(typeof(DeviceInfo)));
483 
484  try
485  {
486  var result = NativeMethods.QueryDevice(id, ptr);
487 
488  if (!result)
489  throw new NativeCallException("QueryDevice", result);
490 
491  if (ptr == IntPtr.Zero)
492  throw new ColoreException("Device query failed, ptr NULL.");
493 
494  var info = (DeviceInfo)Marshal.PtrToStructure(ptr, typeof(DeviceInfo));
495 
496  return info;
497  }
498  finally
499  {
500  Marshal.FreeHGlobal(ptr);
501  }
502  }
