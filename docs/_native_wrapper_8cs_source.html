227  internal static DeviceInfo QueryDevice(Guid id)
228  {
229  var ptr = Marshal.AllocHGlobal(Marshal.SizeOf(typeof(DeviceInfo)));
230 
231  try
232  {
233  var result = NativeMethods.QueryDevice(id, ptr);
234 
235  if (!result)
236  throw new NativeCallException("QueryDevice", result);
237 
238  if (ptr == IntPtr.Zero)
239  throw new ColoreException("Device query failed, ptr NULL.");
240 
241  var info = (DeviceInfo)Marshal.PtrToStructure(ptr, typeof(DeviceInfo));
242 
243  return info;
244  }
245  finally
246  {
247  Marshal.FreeHGlobal(ptr);
248  }
249  }
