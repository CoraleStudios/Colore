505  internal static DeviceInfo QueryDevice(Guid id)
506  {
507  var ptr = Marshal.AllocHGlobal(Marshal.SizeOf(typeof(DeviceInfo)));
508 
509  try
510  {
511  var result = NativeMethods.QueryDevice(id, ptr);
512 
513  if (!result)
514  throw new NativeCallException("QueryDevice", result);
515 
516  if (ptr == IntPtr.Zero)
517  throw new ColoreException("Device query failed, ptr NULL.");
518 
519  var info = (DeviceInfo)Marshal.PtrToStructure(ptr, typeof(DeviceInfo));
520 
521  return info;
522  }
523  finally
524  {
525  Marshal.FreeHGlobal(ptr);
526  }
527  }
